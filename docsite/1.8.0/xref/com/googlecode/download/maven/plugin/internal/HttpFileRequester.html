172         public HttpFileRequester build() throws MojoExecutionException {
173             final HttpFileRequester instance = new HttpFileRequester();
174             instance.uri = requireNonNull(this.uri);
175             instance.progressReport = this.progressReport;
176             instance.connectTimeout = this.connectTimeout;
177             instance.socketTimeout = this.socketTimeout;
178             instance.redirectsEnabled = this.redirectsEnabled;
179             instance.preemptiveAuth = this.preemptiveAuth;
180 
181             instance.credentialsProvider = new BasicCredentialsProvider();
182             if (isNotBlank(this.serverId)) {
183                 requireNonNull(this.mavenSession);
184 
185                 if (this.log.isDebugEnabled()) {
186                     this.log.debug("providing custom authentication for " + this.serverId);
187                 }
188                 final Server server = this.mavenSession.getSettings().getServer(serverId);
189                 if (server == null) {
190                     throw new MojoExecutionException(String.format("Server %s not found", serverId));
191                 }
192                 if (this.log.isDebugEnabled()) {
193                     this.log.debug(String.format("serverId %s supplies username: %s and password: ***",
194                             serverId, server.getUsername()));
195                 }
196                 instance.credentialsProvider.setCredentials(
197                         new AuthScope(this.uri.getHost(), this.uri.getPort()),
198                         new UsernamePasswordCredentials(server.getUsername(), server.getPassword()));
199             } else if (isNotBlank(this.username)) {
200                 if (this.log.isDebugEnabled()) {
201                     this.log.debug("providing custom authentication");
202                     this.log.debug("username: " + username + " and password: ***");
203                 }
204                 instance.credentialsProvider.setCredentials(
205                         new AuthScope(this.uri.getHost(), this.uri.getPort()),
206                         new UsernamePasswordCredentials(this.username, this.password));
207             }
208 
209             if (isNotBlank(this.proxyHost)) {
210                 // TODO: authenticate with the proxy
211                 final HttpHost host = new HttpHost(this.proxyHost, this.proxyPort);
212                 instance.routePlanner = new DefaultProxyRoutePlanner(host);
213                 if (isNotBlank(this.proxyUserName) && isNotBlank(this.proxyPassword)) {
214                     instance.credentialsProvider.setCredentials(
215                             new AuthScope(host.getHostName(), host.getPort()),
216                             isNotBlank(this.proxyNtlmHost) && isNotBlank(this.proxyNtlmDomain)
217                                 ? new NTCredentials(this.proxyUserName, this.proxyPassword, this.proxyNtlmHost,
218                                     this.proxyNtlmDomain)
219                                 : new UsernamePasswordCredentials(this.proxyUserName, this.proxyPassword));
220                 }
221             } else {
222                 instance.routePlanner = new SystemDefaultRoutePlanner(ProxySelector.getDefault());
223             }
224 
225             return instance;
226         }

230     /**
231      * Downloads the resource with the given URI to the specified local file system location.
232      *
233      * @param outputFile the output file
234      * @param headers list of headers
235      */
236     public void download(final File outputFile, List<Header> headers) throws IOException {
237         try (final CloseableHttpClient httpClient = createHttpClientBuilder().build()) {
238             final HttpCacheContext clientContext = HttpCacheContext.create();
239             clientContext.setCredentialsProvider(this.credentialsProvider);
240 
241             if (this.preemptiveAuth) {
242                 final AuthCache authCache = new BasicAuthCache();
243                 authCache.put(new HttpHost(this.uri.getHost(), this.uri.getPort()), new BasicScheme());
244                 clientContext.setAuthCache(authCache);
245             }
246 
247             final HttpGet httpGet = new HttpGet(this.uri);
248             headers.forEach(httpGet::setHeader);
249             httpClient.execute(httpGet, response -> handleResponse(this.uri, outputFile, response),
250                     clientContext);
251         }
252     }
253 
254     /**
255      * Handles response from the server
256      * @param uri request uri
257      * @param outputFile output file for the download request
258      * @param response response from the server
259      * @return original response object
260      * @throws IOException thrown if I/O operations don't succeed
261      */
262     private Object handleResponse( URI uri, File outputFile, HttpResponse response )
263             throws IOException {
264         if (response.getStatusLine().getStatusCode() >= 400) {
265             throw new DownloadFailureException(response.getStatusLine().getStatusCode(),
266                     response.getStatusLine().getReasonPhrase());
267         }
268         if (response.getStatusLine().getStatusCode() >= 301 && response.getStatusLine().getStatusCode() <= 303) {
269             throw new DownloadFailureException(response.getStatusLine().getStatusCode(),
270                     response.getStatusLine().getReasonPhrase()
271                             + ". Not downloading the resource because followRedirects is false.");
272         }
273         final HttpEntity entity = response.getEntity();
274         if (entity != null) {
275             progressReport.initiate( uri, entity.getContentLength());
276             byte[] tmp = new byte[8 * 11024];
277             try (InputStream in = entity.getContent(); OutputStream out =
278                     Files.newOutputStream( outputFile.toPath())) {
279                 int bytesRead;
280                 while ((bytesRead = in.read(tmp)) != -1) {
281                     out.write(tmp, 0, bytesRead);
282                     progressReport.update(bytesRead);
283                 }
284                 out.flush();
285                 progressReport.completed();
286 
287             } catch (IOException ex) {
288                 progressReport.error(ex);
289                 throw ex;
290             }
291         }
292         return entity;
293     }

295     private HttpClientBuilder createHttpClientBuilder() {
296         return HttpClients.custom()
297                         .setDefaultCredentialsProvider(this.credentialsProvider)
298                         .setRoutePlanner(routePlanner)
299                         .setDefaultRequestConfig(RequestConfig.custom()
300                                 .setConnectTimeout(connectTimeout)
301                                 .setSocketTimeout(socketTimeout)
302                                 .setRedirectsEnabled(redirectsEnabled)
303                                 .build());
304     }
