88      private <T> void setVariableValueToObject(Object object, String variable, T value) {
89          try {
90              Field field = ReflectionUtils.getFieldByNameIncludingSuperclasses(variable, object.getClass());
91              field.setAccessible(true);
92              field.set(object, value);
93          } catch (IllegalAccessException e) {
94              throw new RuntimeException(e);
95          }
96      }

278     /**
279      * Verifies that a concurrent invocation of two mojos with, the resulting cache index will keep note of both files.
280      * One of the processes starts first, then gets held up while another process downloads another file. Once
281      * the second process finishes, the first process saves its file as well. The test verifies that both files
282      * are present in the cache index. Finally, the test also verifies that the content of the cached files matches
283      * the "downloaded" content.
284      *
285      * @throws Exception should any exception be thrown
286      */
287     @Test
288     public void testCacheRetainingValuesFromTwoConcurrentCalls() throws Exception {
289         final URI firstMojoUri = URI.create("http://test/foo");
290         final URI secondMojoUri = URI.create("http://test/bar");
291         WGetMojo firstMojo = createMojo(mojo -> {
292             setVariableValueToObject(mojo, "uri", firstMojoUri);
293             setVariableValueToObject(mojo, "outputFileName", OUTPUT_FILE_NAME);
294         });
295         WGetMojo secondMojo = createMojo(mojo -> {
296             setVariableValueToObject(mojo, "uri", secondMojoUri);
297             setVariableValueToObject(mojo, "outputFileName", "second-output-file");
298         });
299 
300         CountDownLatch firstMojoStarted = new CountDownLatch(1), secondMojoFinished = new CountDownLatch(1);
301         Arrays.asList(CompletableFuture.runAsync(() -> {
302                     HttpClientBuilder clientBuilder = createClientBuilder(() -> {
303                         firstMojoStarted.countDown();
304                         try {
305                             // waiting till we're sure the second mojo has finished
306                             assert secondMojoFinished.await(1, SECONDS);
307                         } catch (InterruptedException e) {
308                             throw new RuntimeException(e);
309                         }
310                         return "foo";
311                     });
312                     try (MockedStatic<HttpClientBuilder> builder = mockStatic(HttpClientBuilder.class)) {
313                         builder.when(HttpClientBuilder::create).thenReturn(clientBuilder);
314                         firstMojo.execute();
315                     } catch (MojoExecutionException | MojoFailureException e) {
316                         throw new RuntimeException(e);
317                     }
318                 }),
319                 CompletableFuture.runAsync(() -> {
320                     HttpClientBuilder clientBuilder = createClientBuilder(() -> "bar");
321                     try (MockedStatic<HttpClientBuilder> builder = mockStatic(HttpClientBuilder.class)) {
322                         builder.when(HttpClientBuilder::create).thenReturn(clientBuilder);
323                         // waiting till we're sure the first mojo has started
324                         assert firstMojoStarted.await(1, SECONDS);
325                         secondMojo.execute();
326                         secondMojoFinished.countDown();
327                     } catch (MojoExecutionException | MojoFailureException | InterruptedException e) {
328                         throw new RuntimeException(e);
329                     }
330                 })).forEach(CompletableFuture::join);
331 
332         // cache should contain both entries
333         try (ObjectInputStream stream = new ObjectInputStream(Files.newInputStream(
334                 cacheDirectory.resolve("index.ser")))) {
335             @SuppressWarnings("unchecked")
336             Map<URI, String> index = (Map<URI, String>) stream.readObject();
337             assertThat(index.entrySet(), hasSize(2));
338             assertThat(index.keySet(), containsInAnyOrder(firstMojoUri, secondMojoUri));
339 
340             assertThat(String.join("", Files.readAllLines(cacheDirectory.resolve(index.get(firstMojoUri)))),
341                     is("foo"));
342 
343             assertThat(String.join("", Files.readAllLines(cacheDirectory.resolve(index.get(secondMojoUri)))),
344                     is("bar"));
345         }
346     }

551                 final Throwable rootCause = getRootCause(ex);
552                 if (rootCause.getMessage() == null || !rootCause.getMessage().contains("Not same digest as expected")) {

556             }

559     private static Throwable getRootCause(Throwable t) {
560         while (t.getCause() != null) {
561             t = t.getCause();
562         }
563         return t;
564     }
